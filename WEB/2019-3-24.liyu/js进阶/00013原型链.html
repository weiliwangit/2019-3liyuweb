function Fn(name){
           this.name=name;
}
Fn.prototype.sayName=function(){
           console.log('I am '+"tom");
}

var p1=new Fn("tom");
console.log("p1",p1);

var p2=new Fn("leo");
console.log("p2",p2);

console.log( "Fn.prototype",Fn.prototype);

说明p1和p2的__proto_同时指向Fn.prototype

console.log('Fn.prototype.__proto__',Fn.prototype.__proto__);

console.log( "Object.prototype",Object.prototype)

说明Fn.prototype.__proto__指向Object.prototype

console.log( "Object.prototype.__proto__", Object.prototype.__proto__);

说明Object.prototype.__proto__指向null

综上所述：当用"new 构造函数"创建一个对象实例后,在构造函数中定义的属性和方法会成为对象实例自己的属性和方法,
